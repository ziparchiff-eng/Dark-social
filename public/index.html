async function uploadContent(type) { const username = document.getElementById('username').value || 'Аноним'; const text = document.getElementById('postText').value; const fileInput = document.getElementById('fileInput'); const formData = new FormData(); formData.append('username', username); formData.append('text', text); formData.append('type', type); if (fileInput.files[0]) formData.append('file', fileInput.files[0]); try { const response = await fetch('/upload', { method: 'POST', body: formData }); const result = await response.json(); if (result.success) { document.getElementById('postText').value = ''; document.getElementById('fileInput').value = ''; alert("Опубликовано!"); } else { alert("Ошибка сервера: " + result.error); } } catch (e) { alert("Сетевая ошибка: сервер не ответил"); console.error(e); } }
